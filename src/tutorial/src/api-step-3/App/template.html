📋 完整的 try/catch 代码（背下来） 
try {
  await axios.get(url)
} catch (err) {
  if (err.response) {
    // 形态 ①：后端给了非 2xx 响应
    error.value = err.response.data?.message
               || `HTTP ${err.response.status}`
  } else if (err.request) {
    // 形态 ②：请求发出去但没收到响应
    error.value = '网络错误：后端没启动或被防火墙拦了'
  } else {
    // 形态 ③：配置错误
    error.value = `请求配置错误：${err.message}`
  }
} finally {
  loading.value = false
}
